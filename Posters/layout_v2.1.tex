\newcommand{\mainbox}[2][2]{
	\psframebox[linewidth=0.5mm,linecolor=gray,framearc=0.0000055,
	framesep=0cm]{\noindent
	\psframebox[linewidth=3.5mm,fillstyle=solid,
	fillcolor=lightfill,linecolor=framecolor,framearc=0.0,
	framesep=1cm]{\begin{minipage}{37.5cm}
 	\begin{center}
	\parbox{0.8\textwidth}{\begin{center}{\huge \textcolor{sectiontitles}{#1}}\end{center}}
	\end{center}\vspace{0.75cm}
	#2
	\end{minipage}}}\\\vspace{0.25cm}\\
	}

\newcommand{\equationbox}[1]{
	\vspace{0.5cm}\begin{center}
	$\;
	\tikz[baseline]{
         \node[fill=equationbox,anchor=base](t1){$\displaystyle
            {#1}
         $};}$
	\end{center}\vspace{0.5cm}
	}

				\fi}

							}
